Client-Side Vue SFC Compiler

Script that enables usage of Vue.js Single File Components directly in browser. 

			
				<script src="./vue-compiler.js"></script>

				<script>
					VueCompiler.import([
						'./btn.vue', './dropdown-divider.vue', './dropdown-header.vue', './dropdown-item.vue',
					]);
				</script>

				<html>
					<btn fas="cog" btn-style="success">
						Button
						<template #dropdown>
							<dropdown-header>Header</dropdown-header>
							<dropdown-item @click="font = 1">First</dropdown-item>
							<dropdown-item @click="font = 2">Second</dropdown-item>
							<dropdown-divider></dropdown-divider>
							<dropdown-item @click="font = 0">Third</dropdown-item>
						</template>
					</btn>
				</html>
